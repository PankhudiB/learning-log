I was aware of normal switch statements and type assertion
But came across Type Switches in golang :

```go

func main() {
    do(3)
    do("some-string")
}

func do(i interface{}) {
	switch v := i.(type) {
	case int:
		fmt.Println(fmt.Sprintf("Int : %d", v))
	case string:
		fmt.Println(fmt.Sprintf("Str len : %d-%s", len(v), v))
	}
}

```
